CREATE TABLE Tipo_Item (
  id_Tipo_Item NUMBER(4),
  nm_Tipo_Item VARCHAR(45),
  PRIMARY KEY (id_Tipo_Item)
);

CREATE TABLE Tipo_Habilidade (
  id_Tipo_Habilidade NUMBER(4),
  nm_Tipo_Habilidade VARCHAR2(45),
  PRIMARY KEY(id_Tipo_Habilidade)
);

CREATE TABLE TIpo_Personagem (
  id_Tipo_Personagem NUMBER(4),
  nm_Tipo_Personagem VARCHAR2(45) NULL,
  PRIMARY KEY(id_Tipo_Personagem)
);

CREATE TABLE Usuario (
  id_Usuario NUMBER(4),
  nm_Login VARCHAR2(30) NOT NULL,
  nm_Senha VARCHAR2(15) NOT NULL,
  nm_Pessoa VARCHAR2(45) NOT NULL,
  dt_Nascimento DATE NOT NULL,
  nm_Email VARCHAR2(30) NOT NULL,
  nm_Pergunta VARCHAR2(40) NOT NULL,
  nm_Resposta VARCHAR2(40) NOT NULL,
  dt_Criacao DATE NOT NULL,
  dt_Exclusao DATE NULL,
  vl_Status NUMBER(1) NOT NULL,
  PRIMARY KEY(id_Usuario)
);

CREATE TABLE Tipo_Quest (
  idTipo_Quest NUMBER(4),
  ds_TipoQuest VARCHAR2(45) NULL,
  PRIMARY KEY(idTipo_Quest)
);

CREATE TABLE Raca (
  id_Raca NUMBER(4),
  nm_Raca VARCHAR2(15) NOT NULL,
  ds_Raca VARCHAR2(300) NOT NULL,
  vl_Vida_Base NUMBER(5) NOT NULL,
  vl_Mana_Base NUMBER(5) NOT NULL,
  vl_Stamina_Base NUMBER(5) NOT NULL,
  vl_Forca_Raca NUMBER(5) NOT NULL,
  vl_Inteligencia_Raca NUMBER(5) NOT NULL,
  vl_Destreza_Raca NUMBER(5) NOT NULL,
  vl_Agilidade_Raca NUMBER(5) NOT NULL,
  vl_Sorte_Raca NUMBER(5) NOT NULL,
  vl_Status NUMBER(5) NOT NULL,
  PRIMARY KEY(id_Raca)
);

CREATE TABLE Item (
  id_Item NUMBER(4),
  Classe_id_Classe NUMBER(5),
  Personagem_id_Personagem NUMBER(5),
  Quest_id_Quest NUMBER(5),
  Tipo_Item_id_Tipo_Item NUMBER(5) NOT NULL,
  nm_Item VARCHAR2(20) NOT NULL,
  ds_Item VARCHAR2(60) NOT NULL,
  vl_Ataque NUMBER(5) NULL,
  vl_Defesa NUMBER(5) NULL,
  vl_Durabilidade NUMBER(5) NULL,
  vl_Peso NUMBER(5) NULL,
  vl_Valor NUMBER(5) NULL,
  vl_Level_Minimo NUMBER(5) NOT NULL,
  PRIMARY KEY(id_Item),
  CONSTRAINT FK_Classe_Item FOREIGN KEY(Classe_id_Classe)references Classe(id_classe),
  CONSTRAINT FK_Personagem_Item FOREIGN KEY(Personagem_id_Personagem)references Personagem(id_Personagem),
  CONSTRAINT FK_Quest_Item FOREIGN KEY(Quest_id_Quest)references Quest(id_Quest),
  CONSTRAINT FK_TipoItem_Item FOREIGN KEY(Tipo_Item_id_Tipo_Item)references Tipo_Item(id_Tipo_Item)
);

CREATE TABLE Classe (
  id_Classe NUMBER(4),
  Raca_id_Raca NUMBER(4) NOT NULL,
  vl_Bonus_Forca NUMBER(5) NULL,
  vl_Bonus_Inteligencia NUMBER(5) NULL,
  vl_Bonus_Destreza NUMBER(5) NULL,
  vl_Bonus_Agilidade NUMBER(5) NULL,
  nm_Classe VARCHAR2(15) NOT NULL,
  ds_Classe VARCHAR2(60) NOT NULL,
  vl_Bonus_sorte NUMBER(5) NULL,
  vl_Status NUMBER(5) NOT NULL,
  PRIMARY KEY(id_Classe),
  CONSTRAINT FK_Raca_Classe FOREIGN KEY(Raca_id_Raca)references Raca(id_Raca)	
);

CREATE TABLE Habilidade (
  id_Habilidade NUMBER(4),
  Raca_id_Raca NUMBER(4) NULL,
  Classe_id_Classe NUMBER(4) NULL,
  TipoHab_id_Tipo_Habilidade NUMBER(4) NOT NULL,
  vl_Tempo_Duracao NUMBER(5) NULL,
  vl_Tempo_Regeneracao NUMBER(5) NULL,
  vl_Level_Minimo NUMBER(5) NOT NULL,
  vl_Custo NUMBER(5) NULL,
  vl_Status NUMBER(1) NOT NULL,
  nm_Habilidade VARCHAR2(20) NOT NULL,
  ds_Habilidade VARCHAR2(60) NOT NULL,
  PRIMARY KEY(id_Habilidade),
  CONSTRAINT FK_TipoHabilidade_Habilidade FOREIGN KEY(TipoHab_id_Tipo_Habilidade)references Tipo_Habilidade(id_Tipo_Habilidade),
  CONSTRAINT FK_Classe_Habilidade FOREIGN KEY(Classe_id_Classe)references Classe(id_Classe),
  CONSTRAINT FK_Raca_Habilidade FOREIGN KEY(Raca_id_Raca)references Raca(id_Raca)
);

CREATE TABLE Personagem (
  id_Personagem NUMBER(4),
  Raca_id_Raca NUMBER(4) NOT NULL,
  TIpoPerso_id_Tipo_Personagem NUMBER(4) NOT NULL,
  Usuario_id_Usuario NUMBER(4) NOT NULL,
  nm_Personagem VARCHAR2(30) NOT NULL,
  nm_Genero VARCHAR2(10) NOT NULL,
  dt_Criacao DATE NOT NULL,
  vl_Experiencia NUMBER(5) NOT NULL,
  vl_Status NUMBER(1) NOT NULL,
  vl_Vida_Inicial NUMBER(5) NOT NULL,
  vl_Mana_Inicial NUMBER(5) NOT NULL,
  vl_Stamina_Inicial NUMBER(5) NOT NULL,
  vl_Dano_Base NUMBER(5) NOT NULL,
  dt_Exclusao DATE NULL,
  PRIMARY KEY(id_Personagem),
  CONSTRAINT FK_Usuario_Personagem FOREIGN KEY(Usuario_id_Usuario)references Usuario(id_Usuario),
  CONSTRAINT FK_TipoPersonagem_Personagem FOREIGN KEY(TIpoPerso_id_Tipo_Personagem)references Tipo_Personagem(id_Tipo_Personagem),
  CONSTRAINT FK_Raca_Personagem FOREIGN KEY(Raca_id_Raca)references Raca(id_Raca)
);

CREATE TABLE Quest (
  id_Quest NUMBER(4),
  Personagem_id_Personagem NUMBER(4) NULL,
  Tipo_Quest_idTipo_Quest NUMBER(4) NOT NULL,
  nm_Quest VARCHAR2(30) NOT NULL,
  ds_Quest VARCHAR2(60) NOT NULL,
  vl_Experiencia NUMBER(5) NOT NULL,
  vl_Tempo_Duracao NUMBER(5) NULL,
  vl_Status NUMBER(5) NOT NULL,
  vl_Level_Minimo NUMBER(5) NULL,
  dt_Expiracao DATE NULL,
  dt_Inicio DATE NULL,
  PRIMARY KEY(id_Quest),
  CONSTRAINT FK_TipoQuest_Quest FOREIGN KEY(Tipo_Quest_idTipo_Quest)references Tipo_Quest(idTipo_Quest),
  CONSTRAINT FK_Personagem_Quest FOREIGN KEY(Personagem_id_Personagem)references Personagem(id_Personagem)
);
